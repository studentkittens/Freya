00073     void BrowserList::on_connection_change(bool server_changed, bool is_connected)
00074     {
00075         /* Jump to Settingstab on disconnect,
00076          * and make other browser insensitive,
00077          * or make them sensitive on connect again
00078          * */
00079         for(Gtk::TreeModel::iterator iter = m_refTreeModel->get_iter("0"); iter; iter++)
00080         {
00081             Gtk::TreeRow row = *iter;
00082             AbstractBrowser * browser = row[m_Columns.m_col_browser];
00083             if(browser != NULL)
00084             {
00085                 if(!is_connected && browser->is_visible() && !browser->needs_connection())
00086                 {
00087                     change_browser(browser);
00088                 }
00089                 else if(browser->needs_connection())
00090                 {
00091                     Gtk::Widget * container = browser->get_container();
00092                     if(container != NULL)
00093                     {
00094                         container->set_sensitive(is_connected);
00095                     }
00096                 }
00097             }
00098         }
00099         mp_PluginListview->set_sensitive(is_connected);
