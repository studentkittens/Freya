00162     void Browser::service_browser_callback(AvahiServiceBrowser *b,
00163             AvahiIfIndex interface,
00164             AvahiProtocol protocol,
00165             AvahiBrowserEvent event, 
00166             const char *name,
00167             const char *type,
00168             const char *domain, 
00169             AVAHI_GCC_UNUSED AvahiLookupResultFlags flags) 
00170     {
00171         /* Necessary since this may block till client is fully connected */
00172         AvahiClient * full_client = avahi_service_browser_get_client(b);
00173 
00174         /* Check event, comments from the avahi documentation. */
00175         switch(event)
00176         {
00177             /* The object is new on the network.*/
00178             case AVAHI_BROWSER_NEW:
00179                 {
00180                     Info("-- NEW: %s %s %s",name,type,domain);
00181                     if(avahi_service_resolver_new(full_client, interface, protocol, name, type, domain,
00182                                 (AvahiLookupFlags)AVAHI_PROTO_INET, (AvahiLookupFlags)0, 
00183                                 Browser::wrap_resolve_callback, this) == NULL)
00184                     {
00185                         gchar * format = g_strdup_printf("Failed to resolve service '%s'",name);
00186                         check_client_error(format);
00187                         g_free(format);
00188                     }
00189                     server_counter++;
00190                     update_status_label(); 
00191                     break;
00192                 }
00193                 /* The object has been removed from the network.*/
00194             case AVAHI_BROWSER_REMOVE:
00195                 {
00196                     Info("-- DEL: %s %s %s",name,type,domain);
00197                     window->server_delete(name);
00198                     server_counter--;
00199                     update_status_label(); 
00200                     break;
00201                 }
00202                 /* One-time event, to notify the user that all entries from the caches have been sent. */
00203             case AVAHI_BROWSER_CACHE_EXHAUSTED:
00204                 {
00205                     break;
00206                 }
00207                 /* One-time event, to notify the user that more records will probably not show up in the near future, i.e.
00208                  * all cache entries have been read and all static servers been queried */
00209             case AVAHI_BROWSER_ALL_FOR_NOW:
00210                 {
00211                     Info("-- Got all domains for now...");
00212                     update_status_label(); 
00213                     break;
00214                 }
00215                 /* Browsing failed due to some reason which can be retrieved using avahi_server_errno()/avahi_client_errno() */
00216             case AVAHI_BROWSER_FAILURE:
00217                 {
00218                     check_client_error("Error while browsing");
00219                     break;
00220                 }
00221             default:
