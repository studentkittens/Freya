00074     Glib::ustring Fortuna::get_fortune(void)
00075     {
00076         Glib::ustring retv = "";
00077 
00078         /* Open a pipe to the fortune command and read from it,
00079          * if it isn't installed we will get a string containing
00080          * "fortune:" - in this case the old text stays.
00081          */
00082         FILE * pipe = NULL;
00083         if((pipe = popen(FORTUNE_COMMAND,"r")))
00084         {
00085             char fortune_buf[FORTUNE_BUF_SIZE] = {0};
00086             int bytes = fread(fortune_buf,1,FORTUNE_BUF_SIZE,pipe);
00087             if(bytes != 0 && !strstr(fortune_buf,"fortune:"))
00088             {
00089                 /* Chomp last string */
00090                 char * last_newline = strrchr(fortune_buf,'\n');
00091                 if(last_newline != NULL)
00092                     last_newline[0] = 0;
00093 
00094                 /* Build markup'd string */
00095                 retv = Glib::Markup::escape_text(fortune_buf);
00096                 retv.insert(0,"<span font='10.5' weight='light'>");
00097                 retv.append("</span>");
00098 
00099                 for(Glib::ustring::iterator iter = retv.begin(); iter != retv.end(); iter++)
00100                 {
00101                     /* Erase newline if next char is not a whitespcae,
00102                      * provides basic formatting for fortunes */
00103                     Glib::ustring::iterator next = iter;
00104                     if(*iter == '\n' && !isspace(*(++next)))
00105                         iter = retv.erase(iter);
00106                 }
00107             }
00108             pclose(pipe);
00109         }
00110         return retv;
00111     }
