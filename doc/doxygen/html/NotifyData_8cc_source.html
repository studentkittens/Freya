00060     Status * NotifyData::recv_status_own(void)
00061     {
00062         struct mpd_status * status = NULL;
00063         struct mpd_connection * conn = NULL;
00064         struct mpd_pair *pair = NULL;
00065         Status * retv_status = NULL;
00066         unsigned long NextSongID = 0;
00067 
00068         if(mp_Conn->is_connected())
00069         {
00070             conn = mp_Conn->get_connection();
00071             status = mpd_status_begin();
00072             if(status != NULL) 
00073             {
00074                 while ((pair = mpd_recv_pair(conn)) != NULL) 
00075                 {
00076                     if(!g_ascii_strcasecmp(pair->name,"nextsongid"))
00077                     {
00078                         NextSongID = g_ascii_strtoll(pair->value,NULL,10);
00079                     }
00080                     else
00081                     {
00082                         mpd_status_feed(status, pair);
00083                     }
00084                     mpd_return_pair(conn, pair);
00085                 }
00086             }
00087 
00088             if(status != NULL)
00089             {
00090                 retv_status = new Status(*status,NextSongID);
00091             }
00092                 
00093             if(mp_NextSong != NULL) {
00094                 delete mp_NextSong;
00095                 mp_NextSong = NULL;
00096             }
00097 
00098             mpd_song * c_next_song = mpd_run_get_queue_song_id(conn,NextSongID);
00099             if(c_next_song != NULL)
00100             {
00101                 mp_NextSong = new Song(*c_next_song);
00102             }
00103             else
00104             {
00105                 mp_Conn->clear_error();
00106             }
00107         }
00108         return retv_status;
00109     }
