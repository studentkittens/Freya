00104     void PlaylistManager::on_cell_edited(const Glib::ustring& path_string,const Glib::ustring& new_text)
00105     {
00106         /* Get the position in the View.. */

00108         Gtk::TreeModel::iterator edited_iter = m_refTreeModel->get_iter(path);
00109         bool is_valid = true;
00110 
00111         /* Lookup if Playlistname already there */
00112         for(Gtk::TreeModel::iterator iter = m_refTreeModel->get_iter("0"); iter; iter++)
00113         {
00114             Gtk::TreeRow row = *iter;
00115             if(row[m_Columns.m_col_name] == new_text && iter != edited_iter)
00116             {
00117                 is_valid = false;
00118                 break;
00119             }
00120         }
00121 
00122         if(!is_valid)
00123         {
00124             /* Start editing again */
00125             mp_TreeView->set_cursor(path, m_PlaylistTreeViewCol, m_PlaylistCellRender, true);
00126             mp_StatusLabel->set_text("Playlist already exist. Choose a different name.");
00127         }
00128         else
00129         {
00130             if(edited_iter)
00131             {
00132                 /* Put the new value in the model */
00133                 Gtk::TreeModel::Row row = *edited_iter;
00134                 Glib::ustring old_name = row[m_Columns.m_col_name];
00135                 if(old_name != new_text)
00136                 {
00137                     MPD::Playlist * PList = row[m_Columns.m_col_plist];
00138                     PList->rename(new_text.c_str());

00140             }
00141             mp_StatusLabel->set_text("");
