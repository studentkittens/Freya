00114     void QueueMerger::add_item(void * pSong)
00115     {
00116         g_assert(pSong);
00117         MPD::Song * new_song = (MPD::Song*)pSong;
00118 
00119         if(needsRefill)

00121             add_row(new_song);

00125             if(!mergeIterIsValid)

00140             add_row(new_song);

00148     void QueueMerger::on_client_update(mpd_idle event, MPD::NotifyData& data)
00149     {
00150         if(event & MPD_IDLE_QUEUE)
00151         {
00152             if(!mergeDisabled)
00153             {
00154                 MPD::Status& status = data.get_status();
00155                 unsigned qu_v = status.get_queue_version();
00156                 unsigned qu_l = status.get_queue_length();
00157 
00158                 /* Do a full refill if requested (e.g. on startup) */
00159                 if((wasReconnected && qu_v > lastPlaylistVersion))
00160                 {
00161                     needsRefill = true;
00162                     Info("Doing full refill of the queue.");
00163                     mp_QueueModel->clear(); 
00164                     mp_Client->fill_queue(*this);
00165 
00166                     /* This is usually only necessary on startup. */
00167                     needsRefill = false;
00168                 }
00169                 else
00170                 {
00171                     /* Merge in changes */
00172                     mergeIterIsValid = false;
00173                     mp_Client->fill_queue_changes(*this,lastPlaylistVersion,mergePos);
00174 
00175                     /* Remove old items at end of list */
00176                     if(qu_l < playlistLength)
00177                         trim(qu_l);
00178                 }
00179 
00180                 lastPlaylistVersion = qu_v;
00181                 playlistLength = qu_l; 
00182             }
00183             mergeDisabled = false;
00184         }
00185     }
