00101     void Writer::message(const char * File, int Line, LOGLEVEL level, const char * fmt, ...) 
00102     {
00103         if(this->m_Logfile != NULL)
00104         {
00105             /* Write all va_args in the tmp_buf,
00106              * allocate memory as necessary */
00107             gchar * tmp_buf = NULL;
00108             va_list params;
00109             va_start(params,fmt);
00110             g_vasprintf(&tmp_buf,fmt,params);
00111             va_end(params);
00112 
00113             /* Write current time in the buffer */
00114             char buffer[TIME_BUF_SIZE] = {0};
00115             get_current_time(buffer);
00116             
00117             /* Convert enum to an approp. string */
00118             bool append_location = false;
00119             const char * title = convert_enum_to_str(level,append_location);
00120 
00121             std::stringstream bufstream;
00122 
00123             /* Add Prefix */
00124             bufstream << "[" << title << " " << buffer << "] ";
00125 
00126             /* Infix the message */
00127             bufstream << tmp_buf;
00128 
00129             /* Do the suffix */
00130             if(append_location)
00131                 bufstream << " (" << Glib::path_get_basename(File) << ":" << Line << ")"; 
00132 
00133             bufstream << std::endl;
00134 
00135             /* Write it to all streams */
00136             std::string format = bufstream.str();
00137             fwrite(format.c_str(),1,format.size(),m_Logfile);
00138             fwrite(format.c_str(),1,format.size(),stderr);
00139             fflush(m_Logfile);
00140 
00141             /* Format is dyn. allocated - free */
00142             g_free(tmp_buf);
