00047 gboolean stdin_io_callback(GIOChannel *source, GIOCondition condition, gpointer data)
00048 {
00049     gboolean  retv = true;
00050 
00051     shell_data * watch_data = (shell_data *)data;
00052     GMainLoop * main_loop = watch_data->loop;
00053     MPD::Client * client = watch_data->client;
00054 
00055     int c = getchar();
00056     switch(c)
00057     {
00058         case 'q': 
00059             {
00060                 /* Quit mainloop */
00061                 g_main_loop_ref(main_loop);
00062                 g_main_loop_quit(main_loop);
00063                 retv = false;
00064                 break;
00065             }
00066         case ':':
00067             {
00068                 char cmd[256] = {0};
00069                 if(fgets(cmd,255,stdin))
00070                 {
00071                     /* Chomp newline */
00072                     char * nl = strrchr(cmd,'\n');
00073                     if(nl != NULL) nl[0] = 0;
00074                     client->send_command(cmd);
00075                 }
00076 
00077                 /* Do not unread newline */
00078                 return true;
00079             }
00080         default:
00081             {
00082                 break;
00083             }
00084     }
00085 
00086     /* Unread characters left on the stream (if any) */
00087     while ((c = getchar()) != EOF && c != '\n'); 
00088 
00089     return retv;
00090 }
